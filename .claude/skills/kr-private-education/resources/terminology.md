## 코드 매핑 예시

```python
# constants/terminology.py

class SchoolType(str, Enum):
    """학교 유형"""
    ELEMENTARY = "elementary"  # 초등
    MIDDLE = "middle"          # 중등
    HIGH = "high"              # 고등


class GradeMapping:
    """학년 매핑"""

    @staticmethod
    def to_display(school_type: SchoolType, grade: int) -> str:
        """시스템 값 → 표시용 변환"""
        prefix = {
            SchoolType.ELEMENTARY: "초",
            SchoolType.MIDDLE: "중",
            SchoolType.HIGH: "고"
        }
        return f"{prefix[school_type]}{grade}"

    @staticmethod
    def from_display(display: str) -> tuple[SchoolType, int]:
        """표시용 → 시스템 값 변환"""
        if display.startswith("초"):
            return SchoolType.ELEMENTARY, int(display[1])
        elif display.startswith("중"):
            return SchoolType.MIDDLE, int(display[1])
        elif display.startswith("고"):
            return SchoolType.HIGH, int(display[1])


class SubjectMapping:
    """과목 매핑"""

    KOREAN_TO_ENGLISH = {
        "국어": "korean",
        "영어": "english",
        "수학": "math",
        "과학": "science",
        "사회": "social_studies",
        "논술": "essay"
    }

    ENGLISH_TO_KOREAN = {v: k for k, v in KOREAN_TO_ENGLISH.items()}
```
